import numpy as np
import scipy.sparse as sp
from scipy.io import loadmat
import os
import argparse
from typing import Optional

def load_mat_sparse(mat_path: str, var_name: str = "A") -> Optional[sp.csr_matrix]:
    try:
        mat_dict = loadmat(mat_path)
        if var_name not in mat_dict:
            raise ValueError(f"Variable '{var_name}' not found in MAT file.")
        
        mat = mat_dict[var_name]
        print(f"Successfully loaded MAT file: {mat_path}")
        print(f"Matrix shape: {mat.shape}, dtype: {mat.dtype}")

        if sp.issparse(mat):
            csr_mat = mat.tocsr()
        else:
            csr_mat = sp.csr_matrix(mat)
        
        if csr_mat.dtype != np.float32:
            csr_mat = csr_mat.astype(np.float32)
            print("Matrix dtype converted to float32")

        nnz = csr_mat.nnz
        size = csr_mat.shape[0] * csr_mat.shape[1]
        print(f"CSR Matrix Info: NNZ={nnz}, Density={nnz/size:.2%}")
        return csr_mat

    except Exception as e:
        print(f"Failed to load MAT file: {str(e)}")
        return None

def save_csr_binary(csr_mat: sp.csr_matrix, output_dir: str):
    os.makedirs(output_dir, exist_ok=True)

    val_path = os.path.join(output_dir, "A_val.bin")
    with open(val_path, "wb") as f:
        f.write(csr_mat.data.tobytes())
    val_size = os.path.getsize(val_path) / 1024 / 1024
    print(f"Saved values: {val_path} ({val_size:.2f} MB)")

    row_ptr_path = os.path.join(output_dir, "A_row_ptr.bin")
    with open(row_ptr_path, "wb") as f:
        row_ptr = csr_mat.indptr.astype(np.int32)
        f.write(row_ptr.tobytes())
    row_ptr_size = os.path.getsize(row_ptr_path) / 1024 / 1024
    print(f"Saved row pointers: {row_ptr_path} ({row_ptr_size:.2f} MB)")

    col_idx_path = os.path.join(output_dir, "A_col_idx.bin")
    with open(col_idx_path, "wb") as f:
        col_idx = csr_mat.indices.astype(np.int32)
        f.write(col_idx.tobytes())
    col_idx_size = os.path.getsize(col_idx_path) / 1024 / 1024
    print(f"Saved column indices: {col_idx_path} ({col_idx_size:.2f} MB)")

def parse_args():
    parser = argparse.ArgumentParser(description="MAT to CSR Binary Converter")
    parser.add_argument(
        "--input", "-i", 
        required=True, 
        help="Input MAT file path"
    )
    parser.add_argument(
        "--output-dir", "-o", 
        default="./csr_output/", 
        help="Output CSR binary directory"
    )
    parser.add_argument(
        "--var-name", "-v", 
        default="A", 
        help="Sparse matrix variable name in MAT file"
    )
    return parser.parse_args()

if __name__ == "__main__":
    args = parse_args()
    print("="*60)
    print("MAT to CSR Binary Converter")
    print(f"Input file: {args.input}")
    print(f"Output dir: {args.output_dir}")
    print(f"Variable name: {args.var_name}")
    print("="*60)

    csr_mat = load_mat_sparse(args.input, args.var_name)
    if csr_mat is None:
        print("Conversion failed. Exiting.")
        exit(1)

    save_csr_binary(csr_mat, args.output_dir)

    print("="*60)
    print("Conversion complete! Generated files:")
    print(f"1. {os.path.join(args.output_dir, 'A_val.bin')}")
    print(f"2. {os.path.join(args.output_dir, 'A_row_ptr.bin')}")
    print(f"3. {os.path.join(args.output_dir, 'A_col_idx.bin')}")
